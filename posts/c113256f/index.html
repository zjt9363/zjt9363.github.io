「Codeforce 527 C」最大连续0+线段树
题意是给定一个矩形，不停地纵向或横向切割，问每次切割后，最大的矩形面积是多少。（）最大矩形面积=最长的长*最宽的宽）

题目链接

https://codeforces.com/problemset/problem/527/C

题解

长宽都是10^5，所以，用01序列表示每个点是否被切割，然后，
最长的长就是长的最长连续0的数量+1
最长的宽就是宽的最长连续0的数量+1
于是用线段树维护最长连续零
对于线段树的每个节点，记录四个信息，l从左便开始连续零的个数，r从右边开始连续零的个数，m此区间最大连续零的个数，all是否整个区间都是0。

代码

`

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll MAX = 1000;
const ll MAXN = 200010;

#define INF 0x3f3f3f3f

#define LINF 0x3f3f3f3f3f3f3f3f

#define mem(a, b) memset(a, b, sizeof(a))
ll aw[MAXN], ah[MAXN], w, h, n, p;
char c;
struct node
{
ll l, r, m;
bool all;
} t[2][MAXN << 2];
void pushup(ll rt, ll id)
{
node &cur = t[id][rt], &lson = t[id][rt << 1], &rson = t[id][rt << 1 | 1];
cur.l = lson.l + (lson.all ? rson.l : 0);
cur.r = rson.r + (rson.all ? lson.r : 0);
cur.m = max(lson.r + rson.l, max(lson.m, rson.m));
cur.all = lson.all && rson.all;
}
void Build(ll l, ll r, ll rt, ll id)
{
if (l == r)
{
t[id][rt].l = t[id][rt].r = t[id][rt].m = t[id][rt].all = 1;
if ((id == 0 && l == w) || (id == 1 && l == h))
t[id][rt].l = t[id][rt].r = t[id][rt].m = t[id][rt].all = 0;
return;
}
ll m = (l + r) >> 1;
Build(l, m, rt << 1, id);
Build(m + 1, r, rt << 1 | 1, id);
pushup(rt, id);
}
void update(ll L, ll l, ll r, ll rt, ll id)
{
if (l == r)
{
t[id][rt].l = t[id][rt].r = t[id][rt].m = t[id][rt].all = 0;
return;
}
ll m = (l + r) >> 1;
L <= m ? update(L, l, m, rt << 1, id) : update(L, m + 1, r, rt << 1 | 1, id);
pushup(rt, id);
}
int main()
{
cin >> w >> h >> n;
mem(aw, 0), mem(ah, 0), mem(t, 0);
aw[0] = ah[0] = 1;
Build(1, w, 1, 0);
Build(1, h, 1, 1);
for (ll i = 0; i < n; i++)
{
cin >> c >> p;
c == ‘H’ ? update(p, 1, h, 1, 1) : update(p, 1, w, 1, 0);
cout << (t[0][1].m + 1) * (t[1][1].m + 1) << endl;
}
return 0;
}
